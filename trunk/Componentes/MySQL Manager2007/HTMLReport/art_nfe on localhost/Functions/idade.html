CREATE DEFINER = 'root'@'localhost' FUNCTION `idade`(data DATE)
   RETURNS varchar(80) CHARSET latin1
   NOT DETERMINISTIC
   CONTAINS SQL
   SQL SECURITY DEFINER
   COMMENT ''
BEGIN
 SET @diaAtual=DAY(CURDATE());
 SET @mesAtual=MONTH(CURDATE());
 SET @anoAtual=YEAR(CURDATE());

 SET @diaData=DAY(data);
 SET @mesData=MONTH(data);
 SET @anoData=YEAR(data);

 SET @difAnos=@anoAtual-@anoData;
 SET @difMeses=@mesAtual-@mesData;
 SET @difDias=@diaAtual-@diaData;

 if (@difDias < 0) then
   SET @difDias=30+@difDias;
   SET @difMeses=@difMeses-1;
 end if;

 if (@difMeses<0) then
   SET @difMeses=12+@difMeses;
   SET @difAnos=@difAnos-1;
 end if;

 RETURN CONCAT(@difAnos,' ano(s) ',@difMeses,' mes(es) e ',@difDias,' dia(s)' );


END;
